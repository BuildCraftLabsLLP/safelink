    **Step 4: ASCII digit scan across ALL rendered HTML.**
    Create and run a validation script:
    ```python
    #!/usr/bin/env python3
    """Scan all rendered HTML for Indic digits in phone number contexts."""
    import os
    import re

    INDIC_DIGIT_RANGES = [
        (0x0966, 0x096F),  # Devanagari
        (0x09E6, 0x09EF),  # Bengali
        (0x0A66, 0x0A6F),  # Gurmukhi
        (0x0AE6, 0x0AEF),  # Gujarati
        (0x0B66, 0x0B6F),  # Oriya
        (0x0BE6, 0x0BEF),  # Tamil
        (0x0C66, 0x0C6F),  # Telugu
        (0x0CE6, 0x0CEF),  # Kannada
        (0x0D66, 0x0D6F),  # Malayalam
    ]

    indic_pattern = ''.join(
        f'{chr(start)}-{chr(end)}'
        for start, end in INDIC_DIGIT_RANGES
    )
    regex = re.compile(f'[{indic_pattern}]')

    violations = []
    for root, dirs, files in os.walk('dist'):
        for fname in files:
            if not fname.endswith('.html'):
                continue
            path = os.path.join(root, fname)
            with open(path, 'r', encoding='utf-8') as f:
                content = f.read()
            # Check for Indic digits near tel: links or phone number patterns
            matches = regex.findall(content)
            if matches:
                violations.append((path, len(matches), matches[:3]))

    if violations:
        print(f"FAIL: {len(violations)} files contain Indic digits:")
        for path, count, samples in violations[:10]:
            print(f"  {path}: {count} Indic digits (e.g., {samples})")
    else:
        print("PASS: No Indic digits found in any rendered HTML")
    ```

    Note: Some Indic digits may legitimately appear in translated PROSE (e.g., "3 दिन" meaning "3 days"). The critical check is that digits near `tel:` links and in emergency number displays are ASCII. If the broad scan finds Indic digits, do a focused scan on `<a href="tel:` contexts only. If THOSE have Indic digits, it's a failure. If Indic digits only appear in prose, that's acceptable.

    // I18N Tests

    // I18N-01: English pages still work (covered by existing tests above)

    // I18N-02: Hindi homepage
    console.log('Testing Hindi homepage (I18N-02)...');
    await page.goto(`${BASE_URL}/hi/`, { waitUntil: 'domcontentloaded' });

    const hiLang = await page.$eval('html', el => el.getAttribute('lang'));
    if (hiLang === 'hi') pass('Hindi page has lang="hi"');
    else fail(`Hindi page lang attribute: "${hiLang}" (expected "hi")`);

    const hiContent = await page.content();
    // Check for Devanagari script (Unicode range 0900-097F)
    const hasDevanagari = /[\u0900-\u097F]/.test(hiContent);
    if (hasDevanagari) pass('Hindi homepage contains Devanagari text');
    else fail('Hindi homepage missing Devanagari text');

    // I18N-11: Phone numbers are ASCII on Hindi page
    const hi112Link = await page.$('a[href="tel:112"]');
    if (hi112Link) {
      const hi112Text = await page.$eval('a[href="tel:112"]', el => el.textContent);
      if (/^\d+$/.test(hi112Text.trim())) pass('Hindi 112 link shows ASCII digits');
      else fail(`Hindi 112 link text not ASCII: "${hi112Text}"`);
    } else {
      fail('Hindi page missing tel:112 link');
    }

    // I18N-03: Tamil state page
    console.log('Testing Tamil Nadu page in Tamil (I18N-03)...');
    await page.goto(`${BASE_URL}/ta/state/tamil-nadu/`, { waitUntil: 'domcontentloaded' });

    const taLang = await page.$eval('html', el => el.getAttribute('lang'));
    if (taLang === 'ta') pass('Tamil page has lang="ta"');
    else fail(`Tamil page lang attribute: "${taLang}" (expected "ta")`);

    const taContent = await page.content();
    const hasTamil = /[\u0B80-\u0BFF]/.test(taContent);
    if (hasTamil) pass('Tamil Nadu page contains Tamil script');
    else fail('Tamil Nadu page missing Tamil script');

    // Check line-height for Tamil (should be 1.8)
    const taLineHeight = await page.$eval('body', el => getComputedStyle(el).lineHeight);
    // line-height could be px value, check it's not 1.6
    if (taContent.includes('line-height:1.8')) pass('Tamil page has line-height 1.8');
    else fail('Tamil page line-height not 1.8');

    // I18N-04 to I18N-10: Verify each language homepage exists and has content
    const langTests = [
      { code: 'te', name: 'Telugu', range: /[\u0C00-\u0C7F]/ },
      { code: 'bn', name: 'Bengali', range: /[\u0980-\u09FF]/ },
      { code: 'mr', name: 'Marathi', range: /[\u0900-\u097F]/ },
      { code: 'kn', name: 'Kannada', range: /[\u0C80-\u0CFF]/ },
      { code: 'ml', name: 'Malayalam', range: /[\u0D00-\u0D7F]/ },
      { code: 'gu', name: 'Gujarati', range: /[\u0A80-\u0AFF]/ },
      { code: 'pa', name: 'Punjabi', range: /[\u0A00-\u0A7F]/ },
    ];

    for (const lt of langTests) {
      console.log(`Testing ${lt.name} homepage (I18N)...`);
      const resp = await page.goto(`${BASE_URL}/${lt.code}/`, { waitUntil: 'domcontentloaded' });
      if (resp.status() === 200) pass(`${lt.name} homepage loads (200)`);
      else fail(`${lt.name} homepage status: ${resp.status()}`);

      const langAttr = await page.$eval('html', el => el.getAttribute('lang'));
      if (langAttr === lt.code) pass(`${lt.name} page has lang="${lt.code}"`);
      else fail(`${lt.name} page lang: "${langAttr}" (expected "${lt.code}")`);

      const content = await page.content();
      if (lt.range.test(content)) pass(`${lt.name} homepage has ${lt.name} script`);
      else fail(`${lt.name} homepage missing ${lt.name} script`);
    }

    // Language switcher test
    console.log('Testing language switcher...');
    await page.goto(`${BASE_URL}/state/maharashtra/`, { waitUntil: 'domcontentloaded' });

    // Check Hindi link in switcher goes to /hi/state/maharashtra/
    const hiSwitcherLink = await page.$eval(
      'a[href="/hi/state/maharashtra/"]',
      el => el.textContent
    ).catch(() => null);
    if (hiSwitcherLink) pass('Language switcher has Hindi link to /hi/state/maharashtra/');
    else fail('Language switcher missing Hindi link for Maharashtra page');

    // Check English link does NOT have /en/ prefix
    const enSwitcherLink = await page.$$eval('a', links =>
      links.find(a => a.href.includes('/en/'))
    );
    if (!enSwitcherLink) pass('No /en/ prefix links in switcher');
    else fail('Switcher has /en/ prefix link (should not exist)');

    // Check active language is highlighted (bold)
    await page.goto(`${BASE_URL}/hi/`, { waitUntil: 'domcontentloaded' });
    const activeLangBold = await page.$eval('b', el => el.textContent).catch(() => '');
    if (activeLangBold.includes('HI')) pass('Hindi page shows HI as bold/active');
    else fail(`Active language indicator not found for Hindi: "${activeLangBold}"`);

    // Guide translation test
    console.log('Testing translated guide page...');
    await page.goto(`${BASE_URL}/hi/guide/cyclone/`, { waitUntil: 'domcontentloaded' });
    const guideContent = await page.content();
    const guideHasDevanagari = /[\u0900-\u097F]/.test(guideContent);
    if (guideHasDevanagari) pass('Hindi cyclone guide has translated content');
    else fail('Hindi cyclone guide missing translated content');

    // Malayalam line-height test (should be 2.0)
    console.log('Testing Malayalam line-height...');
    await page.goto(`${BASE_URL}/ml/`, { waitUntil: 'domcontentloaded' });
    const mlContent = await page.content();
    if (mlContent.includes('line-height:2.0')) pass('Malayalam page has line-height 2.0');
    else fail('Malayalam page line-height not 2.0');
